27def run_python_module(modulename, args, debug=False): 

28 """Run a python module, as though with ``python -m name args...``. 

29 

30 `modulename` is the name of the module, possibly a dot-separated name. 

31 `args` is the argument array to present as sys.argv, including the first 

32 element naming the module being executed. 

33 

34 """ 

35 openfile = None 

36 glo, loc = globals(), locals() 

37 try: 

38 try: 

39 # Search for the module - inside its parent package, if any - using 

40 # standard import mechanics. 

41 if '.' in modulename: 

42 packagename, name = modulename.rsplit('.', 1) 

43 package = __import__(packagename, glo, loc, ['__path__']) 

44 searchpath = package.__path__ 

45 else: 

46 packagename, name = None, modulename 

47 searchpath = None # "top-level search" in imp.find_module() 

48 openfile, pathname, _ = imp.find_module(name, searchpath) 

49 

50 # Complain if this is a magic non-file module. 

51 if openfile is None and pathname is None: 

52 raise NoSource( 

53 'module does not live in a file: %r' % modulename 

54 ) 

55 

56 # If `modulename` is actually a package, not a mere module, then we 

57 # pretend to be Python 2.7 and try running its __main__.py script. 

58 if openfile is None: 

59 packagename = modulename 

60 name = '__main__' 

61 package = __import__(packagename, glo, loc, ['__path__']) 

62 searchpath = package.__path__ 

63 openfile, pathname, _ = imp.find_module(name, searchpath) 

64 except ImportError: 

65 _, err, _ = sys.exc_info() 

66 raise NoSource(str(err)) 

67 finally: 

68 if openfile: 

69 openfile.close() 

70 

71 # Finally, hand the file off to run_python_file for execution. 

72 args[0] = pathname 

73 run_python_file(pathname, args, package=packagename, debug=debug)

76def run_python_file(filename, args, package=None, debug=False): 

77 """Run a python file as if it were the main program on the command line. 

78 

79 `filename` is the path to the file to execute, it need not be a .py file. 

80 `args` is the argument array to present as sys.argv, including the first 

81 element naming the file being executed. `package` is the name of the 

82 enclosing package, if any. 

83 

84 """ 

85 try: 

86 source_file = open_source(filename) 

87 except IOError: 

88 raise NoSource('No file to run: %r' % filename) 

89 with source_file: 

90 source = source_file.read() 

91 

92 # Create a module to serve as __main__ 

93 old_main_mod = sys.modules['__main__'] 

94 main_mod = imp.new_module('__main__') 

95 sys.modules['__main__'] = main_mod 

96 main_mod.__file__ = filename 

97 if package: 

98 main_mod.__package__ = package 

99 main_mod.__builtins__ = builtins 

100 

101 # Set sys.argv and the first path element properly. 

102 old_argv = sys.argv 

103 old_path0 = sys.path[0] 

104 sys.argv = args 

105 if package: 

106 sys.path[0] = '' 

107 else: 

108 sys.path[0] = os.path.abspath(os.path.dirname(filename)) 

109 

110 try: 

111 # We have the source. `compile` still needs the last line to be clean, 

112 # so make sure it is, then compile a code object from it. 

113 if not source or source[-1] != '\n': 

114 source += '\n' 

115 code = compile(source, filename, 'exec') 

116 

117 # Execute the source file. 

118 exec_code_object(code, main_mod.__dict__, debug=debug) 

119 finally: 

120 # Restore the old __main__ 

121 sys.modules['__main__'] = old_main_mod 

122 

123 # Restore the old argv and path 

124 sys.argv = old_argv 

125 sys.path[0] = old_path0
