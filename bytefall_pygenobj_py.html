70def _gen_yf(gen): 

71 """Get another generator object. 

72 

73 Corresponding impl.: 

74 https://github.com/python/cpython/blob/3.5/Objects/genobject.c#L272-L289 

75 """ 

76 yf = None 

77 f = gen.gi_frame 

78 

79 # XXX: `f_lasti` might exceed the length of `f_code.co_code` here, but it 

80 # seems not a possible case in CPython implementation. Maybe we have to 

81 # recheck it. 

82 if f and f.f_lasti < len(f.f_code.co_code): 

83 byte_code = f.f_code.co_code 

84 op = byte_code[f.f_lasti] 

85 if op != dis.opmap['YIELD_FROM']: 

86 return None 

87 yf = f.top() 

88 return yf

91def gen_send_ex(gen, value=None, exc=None): 

92 if gen.gi_running: 

93 # TODO: error message for coroutine 

94 raise ValueError('generator already executing') 

95 if gen._finished: 

96 raise StopIteration 

97 if gen.gi_frame.f_lasti == 0 and value is not None: 

98 raise TypeError("Can't send non-None value to a just-started generator") 

99 gen.gi_frame.stack.append(value) 

100 

101 # Run frame and get returned value instantly, and `gi_running` is True 

102 # only in this duration. 

103 # https://github.com/python/cpython/blob/3.5/Objects/genobject.c#L140-L142 

104 gen.gi_running = True 

105 val = None 

106 vm = get_vm() 

107 try: 

108 val = vm.resume_frame(gen.gi_frame, exc=exc) 

109 finally: 

110 gen.gi_running = False 

111 

112 # NOTE: our implementation is different from CPython here. 

113 # In CPython, `gi_frame.f_stacktop` is used to check whether a generator 

114 # is exhausted or not. 

115 # https://github.com/python/cpython/blob/3.5/Python/ceval.c#L1191 

116 # https://github.com/python/cpython/blob/3.5/Objects/genobject.c#L152 

117 if gen._finished: 

118 raise StopIteration(val) 

119 

120 return val

123def gen_throw(gen, exctype, val=None, tb=None): 

124 if tb and not isinstance(tb, types.TracebackType): 

125 raise TypeError('throw() third argument must be a traceback object') 

126 

127 if isinstance(val, str): # XXX: wrap `val` to an Exception instance 

128 val = exctype(val) 

129 

130 yf = _gen_yf(gen) 

131 ret = None 

132 

133 if yf: 

134 # XXX: cannot make sure `exctype` is actually a type? 

135 if match_exception(exctype, GeneratorExit): 

136 gen.gi_running = True 

137 err = gen_close_iter(yf) 

138 gen.gi_running = False 

139 if err < 0: 

140 try: 

141 val = gen.send(None, exc=GeneratorExit) 

142 except GeneratorExit: 

143 pass 

144 # val = gen.send(None, exc=GeneratorExit) 

145 return val 

146 gen._finished = True 

147 six.reraise(exctype, val, tb) 

148 if isinstance(yf, Generator): 

149 gen.gi_running = True 

150 try: 

151 ret = yf.throw(exctype, val, tb) 

152 finally: 

153 gen._finished = True 

154 gen.gi_running = False 

155 else: 

156 meth = getattr(yf, 'throw', None) 

157 if meth is None: 

158 gen._finished = True 

159 six.reraise(exctype, val, tb) 

160 gen.gi_running = True 

161 ret = meth(exctype, val, tb) 

162 gen.gi_running = False 

163 if ret is None: 

164 val = None 

165 ret = gen.gi_frame.pop() 

166 assert ret == yf 

167 gen.gi_frame.f_lasti += 1 

168 ret = gen.send(val) 

169 return ret 

170 else: 

171 GlobalCache().set('last_exception', (exctype, val, tb)) 

172 try: 

173 val = gen.send(None, exc=exctype) 

174 finally: 

175 gen._finished = True 

176 return val

179def gen_close(gen): 

180 """Interal helper function to close a generator. 

181 

182 Corresponding impl.: 

183 https://github.com/python/cpython/blob/3.5/Objects/genobject.c#L291-L322 

184 """ 

185 yf = _gen_yf(gen) 

186 retval = None 

187 err = 0 

188 exc = None 

189 try: 

190 if yf: 

191 gen.gi_running = True 

192 err = gen_close_iter(yf) 

193 gen.gi_running = False 

194 if err == 0: 

195 GlobalCache().set('last_exception', (GeneratorExit, None, None)) 

196 exc = GeneratorExit 

197 

198 retval = gen.send(None, exc=exc) 

199 if retval is not None: 

200 raise RuntimeError('generator ignored GeneratorExit') 

201 except (StopIteration, GeneratorExit): 

202 # generator is closed normally 

203 gen._finished = True 

204 return None 

205 return -1

295def coroutine(func): 

296 """A decorator to mark coroutines, and this implementation is modified 

297 from `asyncio.coroutines.coroutine`. 

298 

299 The reason why this should be done is that the implementation of 

300 `asyncio.coroutine` relies on native types and functions (e.g. 

301 types.GeneratorType, inspect.isgeneratorfunction, ...). However, some 

302 of those native types and functions are re-implemented by us to run in 

303 virtual machine. Therefore, this should be used to replace the original 

304 one to make `@asyncio.coroutine` work. 

305 """ 

306 # Return directly if given function is a coroutine. 

307 if func.__code__.co_flags & 0x0080: 

308 return func 

309 

310 if isinstance(func, Generator): 

311 coro = func 

312 else: 

313 @functools.wraps(func) 

314 def coro(*args, **kw): 

315 res = func(*args, **kw) 

316 if isfuture(res) or isinstance(res, (Generator, CoroWrapper)): 

317 res = yield from res 

318 elif _AwaitableABC is not None: 

319 try: 

320 await_meth = res.__await__ 

321 except AttributeError: 

322 pass 

323 else: 

324 if isinstance(res, _AwaitableABC): 

325 res = yield from await_meth() 

326 return res 

327 

328 if not _DEBUG: 

329 wrapper = coro 

330 else: 

331 @functools.wraps(func) 

332 def wrapper(*args, **kwds): 

333 w = CoroWrapper(coro(*args, **kwds), func=func) 

334 if w._source_traceback: 

335 del w._source_traceback[-1] 

336 w.__name__ = getattr(func, '__name__', None) 

337 w.__qualname__ = getattr(func, '__qualname__', None) 

338 return w 

339 

340 wrapper._is_coroutine = _is_coroutine # For iscoroutinefunction(). 

341 return wrapper
